function getInviteCodes(ss) {
  var sh = ss.getSheetByName('基本資訊');
  if (!sh) return [];
  var val = sh.getRange(3, 2).getValue();
  return String(val).split(',').map(function(c){ return c.trim(); }).filter(Boolean);
}

function doGet(e) {
  // 邀請碼驗證端點
  if (e.parameter.action === 'auth') {
    try {
      var ss = SpreadsheetApp.openById(e.parameter.id);
      var codes = getInviteCodes(ss);
      var ok = codes.length === 0 || codes.indexOf(e.parameter.code) >= 0;
      return ContentService.createTextOutput(
        JSON.stringify({ ok: ok })
      ).setMimeType(ContentService.MimeType.JSON);
    } catch(err) {
      return ContentService.createTextOutput(
        JSON.stringify({ ok: false, error: err.message })
      ).setMimeType(ContentService.MimeType.JSON);
    }
  }
  // 讀取資料（需驗證邀請碼）
  try {
    var ss = SpreadsheetApp.openById(e.parameter.id);
    var codes = getInviteCodes(ss);
    if (codes.length > 0 && codes.indexOf(e.parameter.code) < 0) {
      return ContentService.createTextOutput(
        JSON.stringify({ error: 'unauthorized' })
      ).setMimeType(ContentService.MimeType.JSON);
    }
    var result = {};
    var names = ['基本資訊','行程','航班','住宿','費用','備忘錄','操作紀錄'];
    names.forEach(function(n){
      var sh = ss.getSheetByName(n);
      if (sh) result[n] = sh.getRange(1,1).getValue() || '{}';
    });
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({ error: err.message })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var p = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(p.id);
    // 驗證邀請碼
    var codes = getInviteCodes(ss);
    if (codes.length > 0 && codes.indexOf(p.code) < 0) {
      return ContentService.createTextOutput('unauthorized');
    }
    Object.keys(p.sheets).forEach(function(name){
      var sh = ss.getSheetByName(name) || ss.insertSheet(name);
      sh.getRange(1,1).setValue(p.sheets[name]);
      sh.getRange(1,2).setValue('↑ JSON備份(勿編輯)');
      if (p.tables && p.tables[name]) {
        var t = p.tables[name];
        if (sh.getLastRow() >= 3)
          sh.getRange(3,1,sh.getLastRow()-2,10).clearContent();
        sh.getRange(3,1,t.length,t[0].length).setValues(t);
      }
    });
    return ContentService.createTextOutput('ok');
  } catch(err) {
    return ContentService.createTextOutput('error: ' + err.message);
  }
}
